    credit_cost = CREDIT_COSTS.get(action, 1)
    
    conn = sqlite3.connect('/var/lib/jarvis/users.db')
    c = conn.cursor()
    
    c.execute("SELECT credits FROM users WHERE id = ?", (user_id,))
    current_credits = c.fetchone()[0]
    
    if current_credits < credit_cost:
        conn.close()
        return jsonify({'error': 'Insufficient credits', 'required': credit_cost, 'available': current_credits}), 402
    
    c.execute("UPDATE users SET credits = credits - ? WHERE id = ?", (credit_cost, user_id))
    c.execute("INSERT INTO usage_log (user_id, action, credits_used) VALUES (?, ?, ?)",
              (user_id, action, credit_cost))
    
    conn.commit()
    conn.close()
    
    return jsonify({
        'success': True,
        'action': action,
        'credits_used': credit_cost,
        'remaining_credits': current_credits - credit_cost
    })

def check_daily_streak(user_id):
    """Verifică streak-ul zilnic"""
    conn = sqlite3.connect('/var/lib/jarvis/users.db')
    c = conn.cursor()
    
    today = datetime.now().date().isoformat()
    
    # Verifică dacă s-a logat azi
    c.execute("SELECT streak FROM daily_logins WHERE user_id = ? AND login_date = ?", (user_id, today))
    if c.fetchone():
        conn.close()
        return
    
    # Găsește ultima logare
    c.execute("SELECT login_date, streak FROM daily_logins WHERE user_id = ? ORDER BY login_date DESC LIMIT 1", (user_id,))
    last = c.fetchone()
    
    yesterday = (datetime.now().date() - timedelta(days=1)).isoformat()
    
    if last and last[0] == yesterday:
        new_streak = last[1] + 1
    else:
        new_streak = 1
    
    c.execute("INSERT INTO daily_logins (user_id, login_date, streak) VALUES (?, ?, ?)", (user_id, today, new_streak))
    
    # Bonusuri streak
    if new_streak == 7:
        add_credits(user_id, ACHIEVEMENTS['active_7days']['credits'], '7day_streak')
        
        # Achievement
        c.execute("SELECT achievement_key FROM user_achievements WHERE user_id = ? AND achievement_key = 'active_7days'", (user_id,))
        if not c.fetchone():
            c.execute("INSERT INTO user_achievements (user_id, achievement_key) VALUES (?, ?)", (user_id, 'active_7days'))
    
    if new_streak == 30:
        add_credits(user_id, ACHIEVEMENTS['active_30days']['credits'], '30day_streak')
        
        c.execute("SELECT achievement_key FROM user_achievements WHERE user_id = ? AND achievement_key = 'active_30days'", (user_id,))
        if not c.fetchone():
            c.execute("INSERT INTO user_achievements (user_id, achievement_key) VALUES (?, ?)", (user_id, 'active_30days'))
    
    conn.commit()
    conn.close()

    """Verifică token-ul JWT"""
    try:
        payload = jwt.decode(token, app.config['JWT_SECRET'], algorithms=['HS256'])
        if payload.get('role') == 'super_admin':
            return True
    except:
        pass
    return False

def admin_required(f):
    """Decorator pentru endpoint-uri doar admin"""
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('Authorization', '').replace('Bearer ', '')
        if not token or not verify_token(token):
            return jsonify({'error': 'Unauthorized. Admin access required.'}), 401
        return f(*args, **kwargs)
    return decorated
